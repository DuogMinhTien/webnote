Cookie trong JS

Set Cookie

Sử dụng để thêm một dữ liệu mới vào cookie cname: key lữu trữ, cvalue: giá trị lưu trữ, exdays: thời gian hết hạn (mặc định là 1 năm) Cách dùng: setCookie ("name","John",23) //Cookie này sẽ tồn tại trong vòng 23 ngày Cách dùng: setCookie ("age",26) //Cookie này sẽ tồn tại trong vòng 365 ngày (1 năm) 
function setCookie(cname, cvalue, exdays=365) {
  const d = new Date();
  d.setTime(d.getTime() + (exdays * 24 * 60 * 60 * 1000));
  let expires = "expires="+d.toUTCString();
  document.cookie = cname + "=" + cvalue + ";" + expires + ";path=/";
}

Get Cookie

Sử dụng để lấy value từ document.cookie cname: key lưu trữ, else_value: nếu key lưu trữ không tồn tại sẽ trả về else_value (mặc địch trả về "") Cách dùng: getCookie ("key","does not exist") 
function getCookie(cname, else_value="") {
  let name = cname + "=";
  let ca = document.cookie.split(";");
  for (let i = 0; i < ca.length; i++) {
    let c = ca[i];
    while (c.charAt(0) == " ") {
      c = c.substring(1);
    }
    if (c.indexOf(name) == 0) {
      return c.substring(name.length, c.length);
    }
  }
  return else_value;
}
